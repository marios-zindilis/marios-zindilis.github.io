Get Zenoss Events from ZenDMD

#!/usr/bin/env python

import Globals
from Products.ZenUtils.ZenScriptBase import ZenScriptBase
DMD = ZenScriptBase(connect=True).dmd

Events = DMD.ZenEventManager.getEventList()
